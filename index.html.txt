        const CalculatorTool = () => {
            const [prevCGPA, setPrevCGPA] = useState('');
            const [prevCredits, setPrevCredits] = useState('');
            const [showCourseNames, setShowCourseNames] = useState(true);
            const [courses, setCourses] = useState([
                { id: 1, name: '', credits: 3, gradePoint: 10.0, gradeLabel: 'Excellent (90-100%)' },
                { id: 2, name: '', credits: 3, gradePoint: 9.0, gradeLabel: 'A (80-89%)' },
                { id: 3, name: '', credits: 4, gradePoint: 8.0, gradeLabel: 'B (70-79%)' },
                { id: 4, name: '', credits: 2, gradePoint: 7.0, gradeLabel: 'C (60-69%)' },
            ]);
            const [gpa, setGpa] = useState(0);
            const [cgpa, setCgpa] = useState(0);
            const [totalCredits, setTotalCredits] = useState(0);
            const [aiAnalysis, setAiAnalysis] = useState(null);
            const [isAnalyzing, setIsAnalyzing] = useState(false);
            const [error, setError] = useState(null);

            useEffect(() => {
                calculateResults();
            }, [courses, prevCGPA, prevCredits]);

            const addCourse = () => {
                const newId = courses.length > 0 ? Math.max(...courses.map(c => c.id)) + 1 : 1;
                setCourses([...courses, { id: newId, name: '', credits: 3, gradePoint: 0.0, gradeLabel: 'Select' }]);
            };

            const removeCourse = (id) => {
                setCourses(courses.filter(course => course.id !== id));
            };

            const updateCourse = (id, field, value) => {
                setCourses(courses.map(course => {
                    if (course.id === id) {
                        if (field === 'grade') {
                            const selectedGrade = GRADE_SCALE.find(g => g.value === parseFloat(value));
                            return { 
                                ...course, 
                                gradePoint: parseFloat(value),
                                gradeLabel: selectedGrade ? selectedGrade.label : 'Custom'
                            };
                        }
                        return { ...course, [field]: value };
                    }
                    return course;
                }));
            };

            const calculateResults = () => {
                let semesterPoints = 0;
                let semesterCredits = 0;
                courses.forEach(course => {
                    const creds = parseFloat(course.credits) || 0;
                    const points = parseFloat(course.gradePoint) || 0;
                    semesterPoints += (creds * points);
                    semesterCredits += creds;
                });
                const currentGPA = semesterCredits === 0 ? 0 : (semesterPoints / semesterCredits);
                setGpa(currentGPA.toFixed(2));
                setTotalCredits(semesterCredits);
                const previousPoints = (parseFloat(prevCGPA) || 0) * (parseFloat(prevCredits) || 0);
                const previousCreds = parseFloat(prevCredits) || 0;
                const totalPointsAll = previousPoints + semesterPoints;
                const totalCreditsAll = previousCreds + semesterCredits;
                const finalCGPA = totalCreditsAll === 0 ? 0 : (totalPointsAll / totalCreditsAll);
                setCgpa(finalCGPA.toFixed(2));
            };

            const resetForm = () => {
                if(window.confirm("Are you sure you want to clear all data?")) {
                    setPrevCGPA('');
                    setPrevCredits('');
                    setCourses([{ id: 1, name: '', credits: 3, gradePoint: 10.0, gradeLabel: 'Excellent (90-100%)' }]);
                    setAiAnalysis(null);
                    setError(null);
                }
            };

            const generateAnalysis = async () => {
                setIsAnalyzing(true);
                setAiAnalysis(null);
                setError(null);
                const apiKey = ""; 
                const courseDetails = courses.map((c, i) => `- ${c.name || `Course ${i + 1}`}: ${c.credits} credits, Grade: ${c.gradeLabel} (${c.gradePoint})`).join('\n');
                const prompt = `You are a friendly and strategic academic advisor. Analyze this student's performance data. CONTEXT: Grading Scale: 10.0 (Excellent) down to 0.0 (Fail). Pass is 5.0. Current Semester GPA: ${gpa} / 10.0. Cumulative CGPA: ${cgpa} / 10.0. Total Credits this Semester: ${totalCredits}. COURSES: ${courseDetails}. TASK: Provide a concise, motivating response in Markdown. Do not use H1 (#) headers. Start with H2 (##). Include: 1. ðŸ“Š Performance Snapshot, 2. ðŸŽ¯ Impact Analysis, 3. ðŸ’¡ Strategic Tips. Keep it under 200 words.`;
                
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
                    });
                    if (!response.ok) throw new Error('Failed to fetch analysis');
                    const data = await response.json();
                    const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
                    if (text) setAiAnalysis(text);
                    else throw new Error('No analysis generated');
                } catch (err) {
                    console.error(err);
                    setError("Couldn't generate analysis. Please try again.");
                } finally {
                    setIsAnalyzing(false);
                }
            };

            return React.createElement('div', { className: "bg-slate-50 py-12 px-4 sm:px-6" },
                React.createElement('div', { className: "max-w-4xl mx-auto space-y-6" },
                    React.createElement('div', { className: "flex flex-col md:flex-row md:items-center justify-between gap-4 pb-6 border-b border-slate-200" },
                        React.createElement('div', null,
                            React.createElement('h2', { className: "text-2xl font-bold text-slate-900" }, "Your Grade Calculator"),
                            React.createElement('p', { className: "text-slate-500 text-sm" }, "Fill in your course details below")
                        ),
                        React.createElement('button', { onClick: resetForm, className: "flex items-center space-x-2 text-slate-500 hover:text-red-500 transition-colors self-start md:self-auto" },
                            React.createElement(RefreshCcw, { className: "w-4 h-4" }),
                            React.createElement('span', null, "Reset Data")
                        )
                    ),
                    React.createElement('div', { className: "grid grid-cols-1 lg:grid-cols-3 gap-6" },
                        React.createElement('div', { className: "lg:col-span-2 space-y-6" },
                            React.createElement('section', { className: "bg-white rounded-xl shadow-sm border border-slate-200 p-6" },
                                React.createElement('h2', { className: "text-lg font-semibold text-slate-800 mb-4 flex items-center" },
                                    React.createElement(Save, { className: "w-5 h-5 mr-2 text-indigo-500" }),
                                    "Previous Semesters (Optional)"
                                ),
                                React.createElement('div', { className: "grid grid-cols-1 sm:grid-cols-2 gap-4" },
                                    React.createElement('div', null,
                                        React.createElement('label', { className: "block text-sm font-medium text-slate-600 mb-1" }, "Current CGPA"),
                                        React.createElement('input', { type: "number", min: "0", max: "10", step: "0.01", placeholder: "e.g. 8.5", className: "w-full p-3 bg-slate-50 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500 outline-none", value: prevCGPA, onChange: (e) => setPrevCGPA(e.target.value) })
                                    ),
                                    React.createElement('div', null,
                                        React.createElement('label', { className: "block text-sm font-medium text-slate-600 mb-1" }, "Credits Earned"),
                                        React.createElement('input', { type: "number", min: "0", placeholder: "e.g. 80", className: "w-full p-3 bg-slate-50 border border-slate-200 rounded-lg focus:ring-2 focus:ring-indigo-500 outline-none", value: prevCredits, onChange: (e) => setPrevCredits(e.target.value) })
                                    )
                                )
                            ),
                            React.createElement('section', { className: "bg-white rounded-xl shadow-sm border border-slate-200 p-6" },
                                React.createElement('div', { className: "flex justify-between items-center mb-4" },
                                    React.createElement('div', { className: "flex items-center space-x-3" },
                                        React.createElement(Calculator, { className: "w-5 h-5 text-indigo-500" }),
                                        React.createElement('h2', { className: "text-lg font-semibold text-slate-800" }, "Current Semester"),
                                        React.createElement('button', { onClick: () => setShowCourseNames(!showCourseNames), className: "flex items-center space-x-1 text-xs font-medium text-slate-400 hover:text-indigo-600 bg-slate-100 px-2 py-1 rounded ml-2" },
                                            showCourseNames ? React.createElement(EyeOff, { className: "w-3 h-3" }) : React.createElement(Eye, { className: "w-3 h-3" }),
                                            React.createElement('span', { className: "hidden sm:inline" }, showCourseNames ? 'Hide Names' : 'Show Names')
                                        )
                                    ),
                                    React.createElement('span', { className: "text-sm bg-indigo-50 text-indigo-700 px-3 py-1 rounded-full font-medium" }, `${courses.length} Courses`)
                                ),
                                React.createElement('div', { className: "space-y-3" },
                                    React.createElement('div', { className: "hidden sm:grid sm:grid-cols-12 gap-4 text-xs font-semibold text-slate-500 uppercase tracking-wider px-2" },
                                        showCourseNames && React.createElement('div', { className: "col-span-5" }, "Course Name"),
                                        React.createElement('div', { className: showCourseNames ? "col-span-2 text-center" : "col-span-3 text-center" }, "Credits"),
                                        React.createElement('div', { className: showCourseNames ? "col-span-4" : "col-span-8" }, "Grade"),
                                        React.createElement('div', { className: "col-span-1" })
                                    ),
                                    courses.map((course) => 
                                        React.createElement('div', { key: course.id, className: "grid grid-cols-1 sm:grid-cols-12 gap-3 sm:gap-4 items-start sm:items-center bg-slate-50 p-3 rounded-lg border border-slate-100 hover:border-indigo-200 transition-all" },
                                            showCourseNames && React.createElement('div', { className: "col-span-1 sm:col-span-5" },
                                                React.createElement('input', { type: "text", placeholder: "Course Name", className: "w-full bg-transparent border-b border-transparent focus:border-indigo-500 focus:bg-white px-2 py-1 outline-none text-slate-800 text-sm", value: course.name, onChange: (e) => updateCourse(course.id, 'name', e.target.value) })
                                            ),
                                            React.createElement('div', { className: `col-span-1 ${showCourseNames ? 'sm:col-span-2' : 'sm:col-span-3'}` },
                                                React.createElement('div', { className: "flex items-center sm:justify-center" },
                                                    React.createElement('span', { className: "sm:hidden text-xs font-medium text-slate-500 w-20" }, "Credits:"),
                                                    React.createElement('input', { type: "number", min: "0", step: "0.5", className: "w-full sm:w-16 bg-white border border-slate-200 rounded px-2 py-1 text-center focus:ring-2 focus:ring-indigo-500 outline-none", value: course.credits, onChange: (e) => updateCourse(course.id, 'credits', parseFloat(e.target.value)) })
                                                )
                                            ),
                                            React.createElement('div', { className: `col-span-1 ${showCourseNames ? 'sm:col-span-4' : 'sm:col-span-8'}` },
                                                React.createElement('div', { className: "flex items-center" },
                                                    React.createElement('span', { className: "sm:hidden text-xs font-medium text-slate-500 w-20" }, "Grade:"),
                                                    React.createElement('select', { className: "w-full bg-white border border-slate-200 rounded px-2 py-1 focus:ring-2 focus:ring-indigo-500 outline-none text-slate-700 text-sm", value: course.gradePoint, onChange: (e) => updateCourse(course.id, 'grade', e.target.value) },
                                                        React.createElement('option', { disabled: true, value: -1 }, "Select"),
                                                        GRADE_SCALE.map(g => React.createElement('option', { key: g.label, value: g.value }, `${g.label} (${g.value})`))
                                                    )
                                                )
                                            ),
                                            React.createElement('div', { className: "col-span-1 flex justify-end sm:justify-center" },
                                                React.createElement('button', { onClick: () => removeCourse(course.id), className: "p-1.5 text-slate-400 hover:text-red-500 rounded-full transition-colors" }, React.createElement(Trash2, { className: "w-4 h-4" }))
                                            )
                                        )
                                    )
                                ),
                                React.createElement('button', { onClick: addCourse, className: "mt-6 w-full py-3 flex items-center justify-center space-x-2 border-2 border-dashed border-indigo-200 rounded-lg text-indigo-600 hover:bg-indigo-50 font-medium transition-all" },
                                    React.createElement(Plus, { className: "w-5 h-5" }), React.createElement('span', null, "Add Course")
                                )
                            )
                        ),
                        React.createElement('div', { className: "lg:col-span-1 space-y-6" },
                            React.createElement('div', { className: "sticky top-20 space-y-6" },
                                React.createElement('div', { className: "bg-gradient-to-br from-indigo-600 to-purple-700 rounded-2xl shadow-xl text-white p-6 relative overflow-hidden" },
                                    React.createElement('div', { className: "absolute top-0 right-0 -mt-4 -mr-4 w-24 h-24 bg-white opacity-10 rounded-full blur-xl" }),
                                    React.createElement('div', { className: "absolute bottom-0 left-0 -mb-4 -ml-4 w-20 h-20 bg-white opacity-10 rounded-full blur-xl" }),
                                    React.createElement('h3', { className: "text-indigo-100 font-medium mb-1" }, "Semester GPA"),
                                    React.createElement('div', { className: "flex items-baseline space-x-1 mb-6" }, React.createElement('span', { className: "text-5xl font-bold" }, gpa), React.createElement('span', { className: "text-indigo-200 text-sm" }, "/ 10.0")),
                                    React.createElement('div', { className: "space-y-4" },

                    )
